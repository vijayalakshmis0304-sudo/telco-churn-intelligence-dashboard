Power BI DAX — Churn Measures
Copy-paste ready for Power BI Desktop
// Churn Rate
Churn Rate % =
  DIVIDE(
    COUNTROWS(FILTER(Customers, Customers[Churn]="Yes")),
    COUNTROWS(Customers)
  )

// High Risk Count
High Risk Customers =
  COUNTROWS(FILTER(Predictions,
    Predictions[churn_probability] >= 0.6))

// At-Risk Revenue
Revenue At Risk =
  SUMX(
    FILTER(Customers,
    Customers[risk_tier] = "High Risk"),
    Customers[MonthlyCharges] * 12)

// Avg Churn Prob by Segment
Avg Churn Prob =
  AVERAGE(Predictions[churn_probability])

// Contract Churn Rate
Contract Churn % =
  DIVIDE(
    CALCULATE([Churn Rate %],
    ALLEXCEPT(Customers,Customers[Contract])),
    1)
